### 1. Create Tooltip Component (if not exists)
```tsx
// components/ui/Tooltip.tsx

import { useState, ReactNode } from 'react';

interface TooltipProps {
  children: ReactNode;
  content: string;
  position?: 'top' | 'bottom' | 'left' | 'right';
}

export function Tooltip({ children, content, position = 'top' }: TooltipProps) {
  const [isVisible, setIsVisible] = useState(false);

  const positionClasses = {
    top: 'bottom-full left-1/2 -translate-x-1/2 mb-2',
    bottom: 'top-full left-1/2 -translate-x-1/2 mt-2',
    left: 'right-full top-1/2 -translate-y-1/2 mr-2',
    right: 'left-full top-1/2 -translate-y-1/2 ml-2'
  };

  const arrowClasses = {
    top: 'top-full left-1/2 -translate-x-1/2 border-t-gray-900 dark:border-t-gray-700 border-x-transparent border-b-transparent',
    bottom: 'bottom-full left-1/2 -translate-x-1/2 border-b-gray-900 dark:border-b-gray-700 border-x-transparent border-t-transparent',
    left: 'left-full top-1/2 -translate-y-1/2 border-l-gray-900 dark:border-l-gray-700 border-y-transparent border-r-transparent',
    right: 'right-full top-1/2 -translate-y-1/2 border-r-gray-900 dark:border-r-gray-700 border-y-transparent border-l-transparent'
  };

  return (
    <div 
      className="relative inline-flex"
      onMouseEnter={() => setIsVisible(true)}
      onMouseLeave={() => setIsVisible(false)}
    >
      {children}
      {isVisible && (
        <div className={`absolute z-50 ${positionClasses[position]}`}>
          <div className="px-2 py-1 text-xs font-medium text-white bg-gray-900 dark:bg-gray-700 rounded shadow-lg whitespace-nowrap">
            {content}
          </div>
          <div className={`absolute w-0 h-0 border-4 ${arrowClasses[position]}`} />
        </div>
      )}
    </div>
  );
}
```

### 2. Create Connection Status Icon Component
```tsx
// components/settings/ConnectionStatusIcon.tsx

import { Link2, Link2Off } from 'lucide-react';
import { Tooltip } from '@/components/ui/Tooltip';

interface ConnectionStatusIconProps {
  isConnected: boolean;
  serviceName: string;
  connectedMessage?: string;
  disconnectedMessage?: string;
}

export function ConnectionStatusIcon({ 
  isConnected, 
  serviceName,
  connectedMessage,
  disconnectedMessage 
}: ConnectionStatusIconProps) {
  const tooltipContent = isConnected 
    ? (connectedMessage || `${serviceName} is connected`)
    : (disconnectedMessage || `${serviceName} is not connected. Click to connect.`);

  return (
    <Tooltip content={tooltipContent} position="top">
      <div className={`
        p-1.5 rounded-full cursor-pointer transition-colors
        ${isConnected 
          ? 'bg-green-100 dark:bg-green-900/30 text-green-600 dark:text-green-400' 
          : 'bg-red-100 dark:bg-red-900/30 text-red-500 dark:text-red-400'
        }
      `}>
        {isConnected ? (
          <Link2 className="w-4 h-4" />
        ) : (
          <Link2Off className="w-4 h-4" />
        )}
      </div>
    </Tooltip>
  );
}
```
