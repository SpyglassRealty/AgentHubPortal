router.post('/connect', async (req, res) => {
  const { yentaId } = req.body;
  
  // Validate UUID format
  const uuidRegex = /^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$/i;
  if (!uuidRegex.test(yentaId)) {
    return res.json({ success: false, message: 'Invalid Yenta ID format' });
  }
  
  try {
    // Test connection with ReZen API
    const response = await fetch(
      `${process.env.REZEN_BASE_URL}/api/v1/transactions/participant/${yentaId}/transactions/CLOSED?pageNumber=0&pageSize=1`,
      {
        headers: {
          'X-API-KEY': process.env.REZEN_API_KEY,
          'Content-Type': 'application/json'
        }
      }
    );
    
    if (!response.ok) {
      return res.json({ success: false, message: 'Invalid Yenta ID or connection failed' });
    }
    
    const data = await response.json();
    
    // Save to database/storage
    // await saveYentaId(userId, yentaId);
    
    res.json({
      success: true,
      message: 'Connected successfully!',
      totalDeals: data.totalCount
    });
    
  } catch (error) {
    res.json({ success: false, message: 'Connection failed. Please try again.' });
  }
});

#### 4c. Get Performance Data Endpoint
```javascript
// GET /api/rezen/performance
// Returns all performance metrics

router.get('/performance', async (req, res) => {
  const yentaId = /* get from user's saved settings */;
  
  if (!yentaId) {
    return res.json({ connected: false });
  }
  
  try {
    // Fetch CLOSED transactions
    const closedResponse = await fetch(
      `${process.env.REZEN_BASE_URL}/api/v1/transactions/participant/${yentaId}/transactions/CLOSED?pageNumber=0&pageSize=200&sortBy=ESCROW_CLOSING_DATE&sortDirection=DESC`,
      {
        headers: {
          'X-API-KEY': process.env.REZEN_API_KEY,
          'Content-Type': 'application/json'
        }
      }
    );
    const closedData = await closedResponse.json();
    
    // Fetch OPEN transactions
    const openResponse = await fetch(
      `${process.env.REZEN_BASE_URL}/api/v1/transactions/participant/${yentaId}/transactions/OPEN?pageNumber=0&pageSize=50`,
      {
        headers: {
          'X-API-KEY': process.env.REZEN_API_KEY,
          'Content-Type': 'application/json'
        }
      }
    );
    const openData = await openResponse.json();
    
    // Calculate metrics
    const currentYear = new Date().getFullYear();
    const now = new Date();
    const twelveMonthsAgo = new Date(now.setMonth(now.getMonth() - 12));
    
    // YTD transactions
    const ytdTransactions = closedData.transactions.filter(t => {
      const closeDate = new Date(t.closedAt);
      return closeDate.getFullYear() === currentYear;
    });
    
    // Last 12 months transactions
    const l12mTransactions = closedData.transactions.filter(t => {
      const closeDate = new Date(t.closedAt);
      return closeDate >= twelveMonthsAgo;
    });
    
    // GCI calculations
    const gciYTD = ytdTransactions.reduce((sum, t) => sum + (t.grossCommission?.amount || 0), 0);
    const gciL12M = l12mTransactions.reduce((sum, t) => sum + (t.grossCommission?.amount || 0), 0);
    const pendingGCI = openData.transactions.reduce((sum, t) => sum + (t.grossCommission?.amount || 0), 0);
    const avgPerDeal = ytdTransactions.length > 0 ? gciYTD / ytdTransactions.length : 0;
    
    // Buyer vs Seller breakdown
    const buyerDeals = ytdTransactions.filter(t => t.listing === false);
    const sellerDeals = ytdTransactions.filter(t => t.listing === true);
    const buyerVolume = buyerDeals.reduce((sum, t) => sum + (t.price?.amount || 0), 0);
    const sellerVolume = sellerDeals.reduce((sum, t) => sum + (t.price?.amount || 0), 0);
    
    // Average sale price
    const totalVolume = ytdTransactions.reduce((sum, t) => sum + (t.price?.amount || 0), 0);
    const avgSalePrice = ytdTransactions.length > 0 ? totalVolume / ytdTransactions.length : 0;
    
    // Average days to close
    const daysToCloseList = ytdTransactions
      .filter(t => t.closedAt && t.firmDate)
      .map(t => {
        const closed = new Date(t.closedAt);
        const contract = new Date(t.firmDate);
        return Math.round((closed - contract) / (1000 * 60 * 60 * 24));
      });
    const avgDaysToClose = daysToCloseList.length > 0 
      ? Math.round(daysToCloseList.reduce((a, b) => a + b, 0) / daysToCloseList.length)
      : 0;
    
    // YoY comparison (same period last year)
    const lastYearStart = new Date(currentYear - 1, 0, 1);
    const lastYearSamePeriod = new Date(currentYear - 1, new Date().getMonth(), new Date().getDate());
    const lastYearYTD = closedData.transactions
      .filter(t => {
        const closeDate = new Date(t.closedAt);
        return closeDate >= lastYearStart && closeDate <= lastYearSamePeriod;
      })
      .reduce((sum, t) => sum + (t.grossCommission?.amount || 0), 0);
    
    const yoyChange = lastYearYTD > 0 ? ((gciYTD - lastYearYTD) / lastYearYTD) * 100 : 0;
    
    // Pending pipeline
    const pendingPipeline = openData.transactions.slice(0, 10).map(t => ({
      address: t.address?.oneLine || 'Address TBD',
      price: t.price?.amount || 0,
      estClose: t.closingDateEstimated,
      gci: t.grossCommission?.amount || 0,
      status: t.lifecycleState?.state || 'OPEN'
    }));
    
    res.json({
      connected: true,
      lastUpdated: new Date().toISOString(),
      gciYTD,
      gciL12M,
      pendingGCI,
      pendingDealsCount: openData.totalCount || 0,
      avgPerDeal,
      buyerSide: {
        count: buyerDeals.length,
        volume: buyerVolume,
        percentage: ytdTransactions.length > 0 ? Math.round((buyerDeals.length / ytdTransactions.length) * 100) : 0
      },
      sellerSide: {
        count: sellerDeals.length,
        volume: sellerVolume,
        percentage: ytdTransactions.length > 0 ? Math.round((sellerDeals.length / ytdTransactions.length) * 100) : 0
      },
      avgSalePrice,
      avgDaysToClose,
      yoyChange: Math.round(yoyChange),
      lastYearYTD,
      pendingPipeline,
      totalClosedYTD: ytdTransactions.length,
      totalClosedAllTime: closedData.totalCount
    });
    
  } catch (error) {
    console.error('Performance API error:', error);
    res.status(500).json({ error: 'Failed to fetch performance data' });
  }
});
```

### PART 5: React Component
```jsx
// pages/MyPerformance.jsx
import React, { useState, useEffect } from 'react';
import './MyPerformance.css';

function MyPerformance() {
  const [isConnected, setIsConnected] = useState(false);
  const [isLoading, setIsLoading] = useState(true);
  const [isConnecting, setIsConnecting] = useState(false);
  const [yentaId, setYentaId] = useState('');
  const [error, setError] = useState('');
  const [performanceData, setPerformanceData] = useState(null);

  useEffect(() => {
    loadPerformanceData();
  }, []);

  const loadPerformanceData = async () => {
    try {
      const response = await fetch('/api/rezen/performance');
      const data = await response.json();
      
      if (data.connected) {
        setIsConnected(true);
        setPerformanceData(data);
      } else {
        setIsConnected(false);
      }
    } catch (err) {
      console.error('Failed to load performance data:', err);
    } finally {
      setIsLoading(false);
    }
  };

  const handleConnect = async () => {
    if (!yentaId.trim()) {
      setError('Please enter your Yenta ID');
      return;
    }

    setIsConnecting(true);
    setError('');

    try {
      const response = await fetch('/api/rezen/connect', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ yentaId: yentaId.trim() })
      });

      const data = await response.json();

      if (data.success) {
        setIsConnected(true);
        loadPerformanceData(); // Reload with actual data
      } else {
        setError(data.message || 'Failed to connect');
      }
    } catch (err) {
      setError('Connection failed. Please try again.');
    } finally {
      setIsConnecting(false);
    }
  };

  const handleDisconnect = async () => {
    if (!window.confirm('Disconnect your ReZen account?')) return;
    
    await fetch('/api/rezen/disconnect', { method: 'POST' });
    setIsConnected(false);
    setPerformanceData(null);
    setYentaId('');
  };

  const formatCurrency = (amount) => {
    return new Intl.NumberFormat('en-US', {
      style: 'currency',
      currency: 'USD',
      minimumFractionDigits: 0,
      maximumFractionDigits: 0
    }).format(amount || 0);
  };

  const formatDate = (dateString) => {
    if (!dateString) return 'TBD';
    return new Date(dateString).toLocaleDateString('en-US', { month: 'short', day: 'numeric' });
  };

  if (isLoading) {
    return <div className="loading">Loading...</div>;
  }

  // NOT CONNECTED - Show connection form
  if (!isConnected) {
    return (
      <div className="performance-page">
        <h1>My Performance</h1>
        
        <div className="connect-card">
          <div className="connect-icon">
          {/* Chart icon - use an SVG or icon library */}
        </div>
          <h2>Connect Your ReZen Account</h2>
          <p>Link your ReZen account to see your GCI, deals, and performance metrics.</p>
          
          <div className="yenta-input-group">
            <label htmlFor="yentaId">Yenta ID</label>
            <input
              type="text"
              id="yentaId"
              value={yentaId}
              onChange={(e) => setYentaId(e.target.value)}
              placeholder="e.g., 0d71597f-e3af-47bd-9645-59fc2910656e"
              disabled={isConnecting}
            />
            <small>Your Yenta ID can be found in your ReZen profile or ask your team leader.</small>
          </div>
          
          {error && <div className="error-message">{error}</div>}
          
          <button 
            className="connect-btn"
            onClick={handleConnect}
            disabled={isConnecting}
          >
            {isConnecting ? 'Connecting...' : 'Connect Account'}
          </button>
        </div>
      </div>
    );
  }

  // CONNECTED - Show performance metrics
  const data = performanceData;
  
  return (
    <div className="performance-page">
      <div className="page-header">
        <h1>My Performance</h1>
        <div className="header-actions">
          <span className="last-updated">
            Last updated: {new Date(data.lastUpdated).toLocaleTimeString([], { hour: 'numeric', minute: '2-digit' })}
          </span>
          <button className="refresh-btn" onClick={loadPerformanceData}>üîÑ Refresh</button>
          <button className="disconnect-btn" onClick={handleDisconnect}>Disconnect</button>
        </div>
      </div>

      {/* GCI Summary Cards */}
      <div className="metrics-grid">
        <div className="metric-card">
          <div className="metric-value">{formatCurrency(data.gciYTD)}</div>
          <div className="metric-label">GCI YTD</div>
          {data.yoyChange !== 0 && (
            <div className={`metric-change ${data.yoyChange >= 0 ? 'positive' : 'negative'}`}>
              {data.yoyChange >= 0 ? '‚Üë' : '‚Üì'} {Math.abs(data.yoyChange)}% YoY
            </div>
          )}
        </div>
        
        <div className="metric-card">
          <div className="metric-value">{formatCurrency(data.gciL12M)}</div>
          <div className="metric-label">GCI L12M</div>
          <div className="metric-sublabel">Last 12 Months</div>
        </div>
        
        <div className="metric-card">
          <div className="metric-value">{formatCurrency(data.pendingGCI)}</div>
          <div className="metric-label">Pending GCI</div>
          <div className="metric-sublabel">({data.pendingDealsCount} deals)</div>
        </div>
        
        <div className="metric-card">
          <div className="metric-value">{formatCurrency(data.avgPerDeal)}</div>
          <div className="metric-label">Avg per Deal</div>
        </div>
      </div>

      {/* Buyer vs Seller Breakdown */}
      <div className="breakdown-grid">
        <div className="breakdown-card">
          <div className="breakdown-title">üè† Buyer Side</div>
          <div className="breakdown-stats">
            <span>{data.buyerSide.count} deals</span>
            <span>{formatCurrency(data.buyerSide.volume)} volume</span>
          </div>
          <div className="progress-bar">
            <div className="progress-fill" style={{ width: `${data.buyerSide.percentage}%` }}></div>
          </div>
          <div className="breakdown-percentage">{data.buyerSide.percentage}%</div>
        </div>
        
        <div className="breakdown-card">
          <div className="breakdown-title">üè∑Ô∏è Seller Side (Listings)</div>
          <div className="breakdown-stats">
            <span>{data.sellerSide.count} deals</span>
            <span>{formatCurrency(data.sellerSide.volume)} volume</span>
          </div>
          <div className="progress-bar">
            <div className="progress-fill seller" style={{ width: `${data.sellerSide.percentage}%` }}></div>
          </div>
          <div className="breakdown-percentage">{data.sellerSide.percentage}%</div>
        </div>
      </div>

      {/* Additional Stats */}
      <div className="stats-row">
        <div className="stat-item">
          <span className="stat-label">Average Sale Price:</span>
          <span className="stat-value">{formatCurrency(data.avgSalePrice)}</span>
        </div>
        <div className="stat-item">
          <span className="stat-label">Avg Days to Close:</span>
          <span className="stat-value">{data.avgDaysToClose} days</span>
        </div>
        <div className="stat-item">
          <span className="stat-label">Closed YTD:</span>
          <span className="stat-value">{data.totalClosedYTD} deals</span>
        </div>
      </div>

      {/* Pending Pipeline Table */}
      {data.pendingPipeline && data.pendingPipeline.length > 0 && (
        <div className="pipeline-section">
          <h3>üìã Pending Pipeline</h3>
          <table className="pipeline-table">
            <thead>
              <tr>
                <th>Address</th>
                <th>Price</th>
                <th>Est. Close</th>
                <th>My GCI</th>
                <th>Status</th>
              </tr>
            </thead>
            <tbody>
              {data.pendingPipeline.map((deal, index) => (
                <tr key={index}>
                  <td>{deal.address}</td>
                  <td>{formatCurrency(deal.price)}</td>
                  <td>{formatDate(deal.estClose)}</td>
                  <td className="gci-cell">{formatCurrency(deal.gci)}</td>
                  <td>
                    <span className={`status-badge ${deal.status.toLowerCase().replace('_', '-')}`}>
                      {deal.status.replace('_', ' ')}
                    </span>
                  </td>
                </tr>
              ))}
            </tbody>
          </table>
        </div>
      )}

      {/* YoY Comparison */}
      {data.yoyChange !== 0 && (
        <div className="yoy-section">
          <h3>üìà Year-over-Year</h3>
          <p className={data.yoyChange >= 0 ? 'positive' : 'negative'}>
            You're {data.yoyChange >= 0 ? 'UP' : 'DOWN'} {Math.abs(data.yoyChange)}% compared to this time last year!
          </p>
          <p className="yoy-comparison">
            {new Date().getFullYear()} YTD: {formatCurrency(data.gciYTD)} vs {new Date().getFullYear() - 1} YTD: {formatCurrency(data.lastYearYTD)}
          </p>
        </div>
      )}
    </div>
  );
}

export default MyPerformance;
```
