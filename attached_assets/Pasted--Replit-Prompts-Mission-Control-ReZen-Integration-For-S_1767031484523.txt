## Mission Control - ReZen Connection Component

Create the ReZenConnectionCard component with full validation and state management.

### Component: ReZenConnectionCard.jsx

```jsx
import React, { useState, useEffect } from 'react';
import './ReZenConnectionCard.css';

function ReZenConnectionCard() {
  const [yentaId, setYentaId] = useState('');
  const [isConnected, setIsConnected] = useState(false);
  const [isLoading, setIsLoading] = useState(false);
  const [error, setError] = useState('');
  const [lastSync, setLastSync] = useState(null);
  const [agentInfo, setAgentInfo] = useState(null);

  // Load saved connection on mount
  useEffect(() => {
    loadSavedConnection();
  }, []);

  const loadSavedConnection = async () => {
    try {
      const response = await fetch('/api/settings/rezen-connection');
      const data = await response.json();
      if (data.connected) {
        setYentaId(data.yentaId);
        setIsConnected(true);
        setLastSync(data.lastSync);
        setAgentInfo(data.agentInfo);
      }
    } catch (err) {
      console.error('Failed to load saved connection:', err);
    }
  };

  const handleConnect = async () => {
    if (!yentaId.trim()) {
      setError('Please enter your Yenta ID');
      return;
    }

    // Basic UUID format validation
    const uuidRegex = /^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$/i;
    if (!uuidRegex.test(yentaId.trim())) {
      setError('Invalid Yenta ID format. It should look like: 0d71597f-e3af-47bd-9645-59fc2910656e');
      return;
    }

    setIsLoading(true);
    setError('');

    try {
      const response = await fetch('/api/settings/rezen-connect', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ yentaId: yentaId.trim() })
      });

      const data = await response.json();

      if (data.success) {
        setIsConnected(true);
        setLastSync(new Date().toISOString());
        setAgentInfo(data.agentInfo);
        setError('');
      } else {
        setError(data.message || 'Failed to connect. Please check your Yenta ID.');
      }
    } catch (err) {
      setError('Connection failed. Please try again.');
    } finally {
      setIsLoading(false);
    }
  };

  const handleDisconnect = async () => {
    if (!window.confirm('Are you sure you want to disconnect your ReZen account?')) {
      return;
    }

    setIsLoading(true);

    try {
      await fetch('/api/settings/rezen-disconnect', { method: 'POST' });
      setIsConnected(false);
      setYentaId('');
      setLastSync(null);
      setAgentInfo(null);
    } catch (err) {
      setError('Failed to disconnect. Please try again.');
    } finally {
      setIsLoading(false);
    }
  };

  const formatLastSync = (dateString) => {
    if (!dateString) return '';
    const date = new Date(dateString);
    return date.toLocaleDateString('en-US', {
      month: 'numeric',
      day: 'numeric',
      year: '2-digit',
      hour: 'numeric',
      minute: '2-digit',
      hour12: true
    });
  };

  return (
    <div className="integration-card">
      <div className="integration-header">
        <div className="integration-logo rezen-logo">
          {/* ReZen Logo or placeholder */}
          <svg viewBox="0 0 40 40" fill="none">
            <rect width="40" height="40" rx="8" fill="#1a1a2e"/>
            <text x="50%" y="55%" textAnchor="middle" fill="white" fontSize="14" fontWeight="bold">RZ</text>
          </svg>
        </div>
        <div>
          <div className="integration-title">Connect Your ReZen Account</div>
        </div>
      </div>

      <p className="integration-description">
        Link your ReZen account to see your transaction data, GCI metrics, and performance insights.
      </p>

      {!isConnected ? (
        <>
          <div className="yenta-id-input">
            <label htmlFor="yentaId">
              Yenta ID
              <span className="info-tooltip" title="Your Yenta ID can be found in your ReZen profile settings or by asking your team leader.">
                â„¹ï¸
              </span>
            </label>
            <input
              type="text"
              id="yentaId"
              value={yentaId}
              onChange={(e) => setYentaId(e.target.value)}
              placeholder="e.g., 0d71597f-e3af-47bd-9645-59fc2910656e"
              disabled={isLoading}
            />
            <p className="input-help">
              Your Yenta ID can be found in your ReZen profile or ask your team leader.
            </p>
          </div>

          {error && <div className="error-message">{error}</div>}

          <button 
            className="connect-btn" 
            onClick={handleConnect}
            disabled={isLoading}
          >
            {isLoading ? 'Connecting...' : 'Connect Account'}
          </button>
        </>
      ) : (
        <>
          {agentInfo && (
            <div className="connected-info">
              <p><strong>Connected as:</strong> {agentInfo.name || 'Agent'}</p>
              <p><strong>Total Closed Deals:</strong> {agentInfo.totalDeals?.toLocaleString() || 'N/A'}</p>
            </div>
          )}

          <div className="connection-status">
            <span className="status-dot connected"></span>
            <span>Connected</span>
            {lastSync && (
              <span className="last-sync">Last sync: {formatLastSync(lastSync)}</span>
            )}
            <button className="disconnect-link" onClick={handleDisconnect}>
              Disconnect
            </button>
          </div>
        </>
      )}
    </div>
  );
}

export default ReZenConnectionCard;
```

### Backend API Endpoints:

```javascript
// routes/api/settings.js

// GET /api/settings/rezen-connection
// Returns current connection status
router.get('/rezen-connection', async (req, res) => {
  const userId = req.user.id; // Assumes auth middleware
  
  try {
    // Get from database
    const settings = await db.query(
      'SELECT rezen_yenta_id, rezen_connected_at, rezen_last_sync FROM user_settings WHERE user_id = $1',
      [userId]
    );
    
    if (settings.rows[0]?.rezen_yenta_id) {
      res.json({
        connected: true,
        yentaId: settings.rows[0].rezen_yenta_id,
        lastSync: settings.rows[0].rezen_last_sync
      });
    } else {
      res.json({ connected: false });
    }
  } catch (err) {
    res.status(500).json({ error: 'Failed to load settings' });
  }
});

// POST /api/settings/rezen-connect
// Validates and saves ReZen connection
router.post('/rezen-connect', async (req, res) => {
  const { yentaId } = req.body;
  const userId = req.user.id;
  
  try {
    // Test the connection with ReZen API
    const testResponse = await fetch(
      `${process.env.REZEN_BASE_URL}/api/v1/transactions/participant/${yentaId}/transactions/CLOSED?pageNumber=0&pageSize=1`,
      {
        headers: {
          'X-API-KEY': process.env.REZEN_API_KEY,
          'Content-Type': 'application/json'
        }
      }
    );
    
    if (!testResponse.ok) {
      return res.json({
        success: false,
        message: 'Invalid Yenta ID or connection failed. Please check your ID and try again.'
      });
    }
    
    const data = await testResponse.json();
    
    // Save to database
```

```javascript
// GET /api/performance/summary
router.get('/summary', async (req, res) => {
  const userId = req.user.id;
  
  // Get user's Yenta ID
  const settings = await db.query(
    'SELECT rezen_yenta_id FROM user_settings WHERE user_id = $1',
    [userId]
  );
  
  const yentaId = settings.rows[0]?.rezen_yenta_id;
  
  if (!yentaId) {
    return res.json({ connected: false });
  }
  
  try {
    // Fetch CLOSED transactions
    const closedResponse = await fetch(
      `${process.env.REZEN_BASE_URL}/api/v1/transactions/participant/${yentaId}/transactions/CLOSED?pageNumber=0&pageSize=100&sortBy=ESCROW_CLOSING_DATE&sortDirection=DESC`,
      {
        headers: {
          'X-API-KEY': process.env.REZEN_API_KEY,
          'Content-Type': 'application/json'
        }
      }
    );
    const closedData = await closedResponse.json();
    
    // Fetch OPEN transactions (pending pipeline)
    const openResponse = await fetch(
      `${process.env.REZEN_BASE_URL}/api/v1/transactions/participant/${yentaId}/transactions/OPEN?pageNumber=0&pageSize=20`,
      {
        headers: {
          'X-API-KEY': process.env.REZEN_API_KEY,
          'Content-Type': 'application/json'
        }
      }
    );
    const openData = await openResponse.json();
    
    // Calculate metrics
    const currentYear = new Date().getFullYear();
    const ytdTransactions = closedData.transactions.filter(t => {
      const closeDate = new Date(t.closedAt);
      return closeDate.getFullYear() === currentYear;
    });
    
    const gciYTD = ytdTransactions.reduce((sum, t) => sum + (t.grossCommission?.amount || 0), 0);
    const pendingGCI = openData.transactions.reduce((sum, t) => sum + (t.grossCommission?.amount || 0), 0);
    
    // Buyer vs Seller breakdown
    const buyerDeals = ytdTransactions.filter(t => !t.listing);
    const sellerDeals = ytdTransactions.filter(t => t.listing);
    
    const buyerVolume = buyerDeals.reduce((sum, t) => sum + (t.price?.amount || 0), 0);
    const sellerVolume = sellerDeals.reduce((sum, t) => sum + (t.price?.amount || 0), 0);
    
    // Average sale price
    const totalVolume = ytdTransactions.reduce((sum, t) => sum + (t.price?.amount || 0), 0);
    const avgSalePrice = ytdTransactions.length > 0 ? totalVolume / ytdTransactions.length : 0;
    
    // Average days to close
    const daysToClose = ytdTransactions
      .filter(t => t.closedAt && t.firmDate)
      .map(t => {
        const closed = new Date(t.closedAt);
        const contract = new Date(t.firmDate);
        return Math.round((closed - contract) / (1000 * 60 * 60 * 24));
      });
    const avgDaysToClose = daysToClose.length > 0 
      ? Math.round(daysToClose.reduce((a, b) => a + b, 0) / daysToClose.length)
      : 0;
    
    res.json({
      connected: true,
      gciYTD,
      gciL12M: closedData.transactions.reduce((sum, t) => sum + (t.grossCommission?.amount || 0), 0),
      pendingGCI,
      pendingDealsCount: openData.totalCount,
      avgPerDeal: ytdTransactions.length > 0 ? gciYTD / ytdTransactions.length : 0,
      buyerSide: {
        count: buyerDeals.length,
        volume: buyerVolume,
        percentage: ytdTransactions.length > 0 ? (buyerDeals.length / ytdTransactions.length) * 100 : 0
      },
      sellerSide: {
        count: sellerDeals.length,
        volume: sellerVolume,
        percentage: ytdTransactions.length > 0 ? (sellerDeals.length / ytdTransactions.length) * 100 : 0
      },
      avgSalePrice,
      avgDaysToClose,
      pendingPipeline: openData.transactions.slice(0, 5).map(t => ({
        address: t.address?.oneLine,
        price: t.price?.amount,
        estClose: t.closingDateEstimated,
        gci: t.grossCommission?.amount,
        status: t.lifecycleState?.state
      })),
      totalClosedYTD: ytdTransactions.length,
      totalClosedAllTime: closedData.totalCount
    });
    
  } catch (err) {
    console.error('Performance API error:', err);
    res.status(500).json({ error: 'Failed to fetch performance data' });
  }
});
```

```jsx
function PendingPipelineTable({ transactions }) {
  const formatCurrency = (amount) => {
    return new Intl.NumberFormat('en-US', {
      style: 'currency',
      currency: 'USD',
      minimumFractionDigits: 0,
      maximumFractionDigits: 0
    }).format(amount);
  };

  const formatDate = (dateString) => {
    if (!dateString) return 'TBD';
    const date = new Date(dateString);
    return date.toLocaleDateString('en-US', { month: 'short', day: 'numeric' });
  };

  const getStatusBadge = (status) => {
    const statusMap = {
      'APPROVED_FOR_CLOSING': { label: 'Pending', color: '#fd7e14' },
      'UNDER_CONTRACT': { label: 'Under Contract', color: '#ffc107' },
      'PRE_OFFER': { label: 'Pre-Offer', color: '#6c757d' },
      'ACTIVE': { label: 'Active', color: '#28a745' }
    };
    const config = statusMap[status] || { label: status, color: '#6c757d' };
    return (
      <span className="status-badge" style={{ background: config.color }}>
        {config.label}
      </span>
    );
  };

  if (!transactions || transactions.length === 0) {
    return (
      <div className="pipeline-empty">
        <p>No pending deals at this time.</p>
      </div>
    );
  }

  return (
    <div className="pipeline-section">
      <div className="pipeline-header">
        <h3>ğŸ“‹ Pending Pipeline</h3>
        <a href="/pipeline" className="view-all">View All â†’</a>
      </div>
      <table className="pipeline-table">
        <thead>
          <tr>
            <th>Property</th>
            <th>Price</th>
            <th>Est. Close</th>
            <th>My GCI</th>
            <th>Status</th>
          </tr>
        </thead>
        <tbody>
          {transactions.map((t, index) => (
            <tr key={index}>
              <td className="property-cell">
                <div className="property-address">{t.address?.split(',')[0]}</div>
                <div className="property-city">{t.address?.split(',').slice(1).join(',')}</div>
              </td>
              <td>{formatCurrency(t.price)}</td>
              <td>{formatDate(t.estClose)}</td>
              <td className="gci-cell">{formatCurrency(t.gci)}</td>
              <td>{getStatusBadge(t.status)}</td>
            </tr>
          ))}
        </tbody>
      </table>
    </div>
  );
}
```
